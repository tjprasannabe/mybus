using UnityEngine; public class BusController : MonoBehaviour { public float accelerationForce = 5000f; public float steeringAngle = 30f; public float brakeForce = 3000f; private Rigidbody rb; void Start() { rb = GetComponent(); } void FixedUpdate() { // Get input from the user float verticalInput = Input.GetAxis("Vertical"); float horizontalInput = Input.GetAxis("Horizontal"); // Apply forward/backward force for acceleration Vector3 force = transform.forward * verticalInput * accelerationForce; rb.AddForce(force); // Apply steering float turn = horizontalInput * steeringAngle; Quaternion turnRotation = Quaternion.Euler(0f, turn, 0f); rb.MoveRotation(rb.rotation * turnRotation); // Simple braking if (Input.GetKey(KeyCode.Space)) { rb.AddForce(-rb.velocity.normalized * brakeForce); } } }
